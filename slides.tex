\begin{frame}
\frametitle{Online Decision Making}

\begin{itemize}
\item Requests $R_1, R_2, \dots, R_T$ arrive, one at a time.
\item For each request $R_t$, we need to choose an action $I_t \in \{1,2,\dots,N\}$.
\item Each action $i$, has a vector of predicted business metrics
$$
(R_{t,i,0}, R_{t,i,1}, \dots, R_{t,i,N}) \in \R^{M+1}
$$
associated with the request $R_t$.
\item In other words, the action $R_t$ is an $N \times (M+1)$ matrix.
\item The sequence of actions $I_1, I_2, \dots, I_T$ should maximize a
certain business metric subject to constraints on other business metrics.
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Linear Program Formulation}

We encode action $I_t$ as a vector $x_t \in \R^N$

\begin{align*}
\maximize_{x_{1,1}, \dots, x_{T,N} \in \R} \quad & \frac{1}{T} \sum_{t=1}^T \sum_{i=1}^N R_{t,i,0} \cdot x_{t,i} \\
\subjectto \quad & \frac{1}{T} \sum_{t=1}^T \sum_{i=1}^N R_{t,i,j} \cdot x_{t,i} \le B_j && \quad 1 \le j \le M \\
& \sum_{i=1}^N x_{t,i} = 1 && \quad 1 \le t \le T \\
& x_{t,i} \ge 0 && \quad 1 \le i \le N, 1 \le t \le T
\end{align*}

We allow randomization.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Policy Optimization}

\begin{itemize}
\item Equivalently, the problem can be formulated as a policy optimization problem.
\item Define the probability simplex
$$
\Delta_N = \left\{ (x_1, x_2, \dots, x_N) \in \R^N ~:~ x_1, x_2, \dots, x_N \ge 0, \sum_{i=1}^N x_i = 1  \right\} \: .
$$
\item Find a policy $\pi: \R^{N \times (M+1)} \to \Delta_N$ such that
\end{itemize}

\begin{align*}
\maximize_{\pi \in \Pi} \quad & \frac{1}{T} \sum_{t=1}^T \sum_{i=1}^N R_{t,i,0} \cdot \pi(R_t)_i \\
\subjectto \quad & \frac{1}{T} \sum_{t=1}^T \sum_{i=1}^N R_{t,i,j} \cdot \pi(R_t)_i \le B_j & 1 \le j \le M
\end{align*}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Distributional Policy Optimization}

\begin{itemize}
\item We don't know the sequence of requests $R_1, R_2, \dots, R_T$ ahead of time.
\item So, we can't solve the linear program or the policy optimization problem directly.
\item We make the assumption that the requests are sampled i.i.d. from a fixed, but unknown, distribution $D$.
\item Find a policy $\pi: \R^{N \times (M+1)} \to \Delta_N$ such that
\end{itemize}

\begin{align*}
\maximize_{\pi \in \Pi} \quad & \Exp_{R \sim D} \left[ \sum_{i=1}^N \pi(R)_i \cdot R_{i,0} \right] \\
\subjectto \quad & \Exp_{R \sim D} \left[ \sum_{i=1}^N \pi(R)_i \cdot R_{i,j} \right] \le B_j & 1 \le j \le M
\end{align*}

\begin{itemize}
\item We assume that we have a historical i.i.d. sample from $D$ available.
\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Lagrangian Policies}

\begin{theorem}[Lagrangian Policy]
Let $D$ be a distribution with a probability density on a bounded subset of
$\R^{N \times (M+1)}$. The distributional policy optimization problem has an
optimal solution of the form $\pi_\lambda:\R^{N \times (M+1)} \to \Delta_N$
given by
$$
\pi_\lambda(R) = e_{i^*} \qquad \text{for any request $R \in \R^{N \times (M+1)}$}
$$
where
\begin{align*}
e_{i^*} & = \underbrace{(0,0,\dots,0,1,0,\dots,0)}_{\text{$1$ is at position  $i^*$}} \in \R^N \\
\lambda & = (\lambda_1, \dots, \lambda_M) \in \R^M \\
i^* & = \argmax_{i=1,2,\dots,N} R_{i,0} - \lambda_1 R_{i,1} - \lambda_2 R_{i,2} - \dots - \lambda_M R_{i,M}
\end{align*}
Tie breaking in argmax is towards the smaller index $i$.
\end{theorem}
\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{How do we find the multipliers $\lambda$?}

We solve the dual program on a historical sample of requests.

\begin{align*}
\minimize_{\substack{\mu_1, \dots, \mu_T \in \R \\ \lambda_1, \dots, \lambda_M \in \R}} \quad & \sum_{t=1}^T \mu_t + \sum_{j=1}^M B_j \lambda_j \\
\subjectto \quad & \mu_t + \frac{1}{T} \sum_{j=1}^M R_{t,i,j} \lambda_j  \ge \frac{1}{T} R_{t,i,0} && \quad 1 \le i \le N, 1 \le t \le T \\
& \lambda_j \ge 0 && \quad 1 \le j \le M
\end{align*}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Lagrangian Policies}

\begin{itemize}
\item Lagrangian policies are the same as multiclass linear classifiers, or, in other words, multiclass Perceptrons.

\item Proof of theorem relies on
\begin{itemize}
\item Uniform convergence for Lagrangian policies.
\item Linear programming duality.
\end{itemize}

\item $\lambda$ from historical data yields a near optimal policy.
\item This follows from the same two arguments:
\begin{itemize}
\item Uniform convergence for Lagrangian policies.
\item Linear programming duality.
\end{itemize}

\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Uniform Convergence for Lagrangian Policies}

\begin{theorem}[Uniform Convergence for Lagrangian Policies]
Let $D$ be a distribution on a bounded subset of $\R^{N \times (M+1)}$. Let $T$
be a positive integer and let $\delta \in (0,1)$. Let $R_1, R_2, \dots, R_T$ be
an i.i.d. sample from $D$. Then, with probability $1-\delta$,
$$
\max_{0 \le j \le M} \ \sup_{\lambda \in \R^M} \ \left| \Exp_{R \sim D} \left[ \sum_{i=1}^N \pi_\lambda(R)_i \cdot R_{i,j} \right] -  \frac{1}{T} \sum_{t=1}^T \sum_{i=1}^N \pi_\lambda(R)_i \cdot R_{t,i,j} \right| \le \epsilon
$$
where $\epsilon = O\left( \sqrt{\frac{M (\ln(T) + \ln N + \ln(1/\delta))}{T}} \right)$.
\end{theorem}

\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Dual of the Linear Program}

\begin{align*}
\minimize_{\substack{\mu_1, \dots, \mu_T \in \R \\ \lambda_1, \dots, \lambda_M \in \R}} \quad & \sum_{t=1}^T \mu_t + \sum_{j=1}^M B_j \lambda_j \\
\subjectto \quad & \mu_t + \frac{1}{T} \sum_{j=1}^M R_{t,i,j} \lambda_j  \ge \frac{1}{T} R_{t,i,0} && \quad 1 \le i \le N, 1 \le t \le T \\
& \lambda_j \ge 0 && \quad 1 \le j \le M
\end{align*}

\begin{itemize}
\item $\mu_1, \mu_2, \dots, \mu_T$ can be computed from the optimal values of $\lambda_1, \dots, \lambda_M$
\item Most values of $x_{1,1}, x_{1,2}, \dots, x_{T,N}$ can be computed from the optimal values of $\lambda_1, \dots, \lambda_M$
\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Complementary slackness}

\begin{lemma}[Complementary slackness]
Suppose $\mu_1, \mu_2, \dots, \mu_T$, $\lambda_1, \dots, \lambda_M$ is the optimal solution of the dual program.
\begin{itemize}
\item If the strict inequality
$$
\mu_t + \frac{1}{T} \sum_{j=1}^M R_{t,i,j} \lambda_j  > \frac{1}{T} R_{t,i,0}
$$
holds, then $x_{t,i} = 0$.

\item In particular, if
$$
i^* = \argmax_{i=1,2,\dots,N} R_{t,i,0} - R_{t,i,1} \lambda_1 - R_{t,i,2} \lambda_2 - \dots - R_{t,i,M} \lambda_M
$$
is unique (i.e. no ties) then $x_{t,i^*} = 1$ and $x_{t,i} = 0$ for all $i \neq i^*$.
\end{itemize}
\end{lemma}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{How many ties are there?}

\begin{itemize}
\item If the requests are in ``general position'', then there at most $M$ requests with a tie in
$$
i^* = \argmax_{i=1,2,\dots,N} R_{t,i,0} - \sum_{j=1}^M R_{t,i,j} \lambda_j \ ?
$$

\item Every tie corresponds to a linear equation for $(\lambda_1, \lambda_2, \dots, \lambda_M)$,
$$
R_{t,i,0} - \sum_{j=1}^M R_{t,i,j} \lambda_j = R_{t,i',0} - \sum_{j=1}^M R_{t,i',j} \lambda_j
$$

\item $M$ variables cannot satisfy $M+1$ linearly independent equations.

\item If the requests are sampled from a distribution with a probability density,
they are in general position with probability one.
\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Errors}

\begin{itemize}
\item We solve the problem on a historical sample of requests.
\item Two sources of error:
\begin{itemize}
\item Sampling error of past data.
\item Sampling error of future data.
\item Lagrangian policy can disagree on $M$ requests with the optimal solution. (Ties)
\end{itemize}

\item Assume that all entries of $R_t$ lie in a bounded interval
\item Sampling errors are $\widetilde{O}(\sqrt{M/T})$.
\item The ties cause error of order $O(M/T)$.
\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Distributions without probability density}

\begin{itemize}
\item If the requests are not in general position, there can be ties.

\item If the distribution does not have a probability density, the optimal
solution might not be a Lagrangian policy.

\item Simple solution: Add a little bit of noise to each request $R_t$:
$$
R'_t = R_t + \epsilon Z_t
$$
where $Z_t$ is a matrix with entries sampled i.i.d. from the uniform distribution on $[-1,1]$
and $\epsilon$ is a small positive constant.

\item Noise is added both to historical and future requests.

\item This causes error of order $O(\epsilon)$.
\end{itemize}
\end{frame}
